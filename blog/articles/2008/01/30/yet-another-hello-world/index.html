using System;
using System.Linq.Expressions;

class HelloWorldOfTrees {

	static void Main ()
	{
		var param = Expression.Parameter (typeof (string), "p");

		var cwl = Expression.Lambda> (
			Expression.Call (
				typeof (Console).GetMethod (
					"WriteLine", new [] { typeof (string) }),
				param), param).Compile ();

		cwl ("Hello World !");
	}
}
