using System;
using System.IO;
using System.Text;

class CompactFrameworkPatcher {

    static readonly byte [] fwPkToken = new byte [] {
        0xb7, 0x7a, 0x5c, 0x56, 0x19, 0x34, 0xe0, 0x89
    };

    static readonly byte [] cfPkToken = new byte [] {
        0x96, 0x9d, 0xb8, 0x05, 0x3d, 0x33, 0x22, 0xac
    };

    static void Main (string [] args)
    {
        if (args.Length != 1) {
            Usage ();
            return;
        }

        string file = args [0];

        using (FileStream fs = new FileStream (
            file, FileMode.Open, FileAccess.ReadWrite)) {

            while (fs.Position + 8 < fs.Length) {
                if (fs.ReadByte () == fwPkToken [0] &&
                    fs.ReadByte () == fwPkToken [1] &&
                    fs.ReadByte () == fwPkToken [2] &&
                    fs.ReadByte () == fwPkToken [3] &&
                    fs.ReadByte () == fwPkToken [4] &&
                    fs.ReadByte () == fwPkToken [5] &&
                    fs.ReadByte () == fwPkToken [6] &&
                    fs.ReadByte () == fwPkToken [7]) {

                    fs.Position -= 8;
                    fs.Write (cfPkToken, 0, 8);
                    fs.Position += 8;
                }
            }
        }

        Console.WriteLine ("Assembly {0} patched", file);
    }

    static void Usage ()
    {
        Console.WriteLine ("Mono Assembly To Compact Framework Patcher");
        Console.WriteLine ("usage: cf-patcher.exe assembly");
    }
}
